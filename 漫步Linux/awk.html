# awk

# 语言结构

awk 'BEGIN {...} pattern { command } END {...}' file
# 脚本通常放在单引号或双引号中
awk 'BEGIN{ i = 0 } { i++ } END { print i }' file
awk "BEGIN{ i = 0 } { i++ } END { print i }" file
# 模式时0后面的动作不执行,!0后面的动作才执行

# 执行流程

  1. 执行GEGIN
  2. 一行一行读取文件内容执行pattern,如果只有pattern,会有默认命令{print $0}
  3. 执行END

# 常用参数

awk -F '[#$]' '{print $NF}' file

# BEGIN模块

在awk读取文件之前,常常用来修改内置变量.有BEGIN可以不需要文件名.

awk 'BEGIN { RS = "#" } { print $0 }' file # 修改内置变量
awk 'BEGIN { print "-------"} { print $0 }' file # 打印分隔行
awk 'BEGIN { a = 8 ; b = 90 ; print a+b , a-b , a/b , a%b }'  # 变量初始化

# END模块

在文件读取完之后,一般用来输出结果.

awk '/^$/' file | wc -l # awk 统计空行
awk '/^$/ {i=i+1} END {print i}' # 输出结果
# sed grep 统计空行
grep '^$' file | wc -l # grep 统计空行
sed -n '/^$/p' file | wc -l # sed 统计空行

# a++ 和 ++a

awk 'BEGIN {print ++a}' # 1

awk 'a[$n]=$0 END { for( i in a ){print a[i]}}' file # 显示文件结尾不重复的行

# 内置变量

变量 含义
NF 字段个数
NR 行号
FS 分隔符
RS 记录分隔符
FNR 当前文件行号,每个文件单独计算
FILENAME 当前文件的文件名

# 正则表达式

正则表达式操作符
~ 记录或区域的表达式进行匹配
!~ 表达不匹配
元字符 功能
^ 开头
$ 结尾
. 任意单字符
* 重复前一个字符0次或n次
+ 重复前一个字符0次或n次
? 重复前一个字符0次或1次
[] 匹配字符组内任意一个字符
[^] 匹配不在字符组内的任意字符
() 子表达式组合
| 或者
{m} 重复m次
{m,} 至少重复m次
{m,n} 重复m次不超过n次

# 案例

# cat t
awk -F '#' 'NR == 1 {print $0}' file # 既有 pattern 又有command
awk -F '#' 'NR == 1' file # 只有pattern,默认print $0
awk -F '#' '{print $0}' file # 只有command,默认所有行
awk -F '#' 'NF ==1 {print $0} NF == 3 { print $(NF-1)}' # 有多个pattern
# FNR 和 NR 类似, FNR 多个文件记录不递增,每个都从0开始
# 正则案例
awk '/321/{print $0}' file
awk '$1~/^B/{print $0}$NF~/1$/' file # 匹配第一列B开头和最后一列1结尾的行,多个模式时默认或关系
awk '$1~/^B|^C/ print $0' file # 匹配第一列B或C开头的行
awk '$1~/^[BC]/ print $0' file # 匹配第一列B或C开头的行
awk '$1~/^(B|C)/ print $0' file # 匹配第一列B或C开头的行
awk '/ABC/,/CBA/ print $0' file # 多个模式用逗号分隔
netstat -n | awk '/^tcp/ {++S[$NF]} END { for (a in S ) print a, s[a]}' # 统计 tcp 各连接数量
awk '{uv[$1]++;next}END{for (ip in uv) print ip,uv[ip]}' # next 关键字跳转到下一行,把下一行作为输入
awk '{print substr($4, 2,11)}' log.log # substr 字符串截取
