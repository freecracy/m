# channel

  1. buf 有缓存 channel 所持有的结构,用来存储缓存数据,是个循环链表
  2. sendx 和 recvx 用于记录 buf 这个循环链表中的发送和接收的 index
  3. lock 是个护持锁
  4. recvq 和 sendq 分别是接收和发送 channel 的 goroutine 抽象出来的结构体 sudog 的队列,是个双向链表

# 创建

make 创建一个 channel 实际是在内存中实例化一个 hchan 结构体,并返回一个 hchan 指针,我们在函数中传递的也是这个指针,所以函数中无需再使用 channel 指针.有缓冲最大1<<16 65535.

// src/runtime/chan.hchan
type hchan struct {
	qcount   uint           // total data in the queue
	dataqsiz uint           // size of the circular queue
	buf      unsafe.Pointer // points to an array of dataqsiz elements
	elemsize uint16
	closed   uint32
	elemtype *_type // element type
	sendx    uint   // send index
	recvx    uint   // receive index
	recvq    waitq  // list of recv waiters
	sendq    waitq  // list of send waiters

	// lock protects all fields in hchan, as well as several
	// fields in sudogs blocked on this channel.
	//
	// Do not change another G's status while holding this lock
	// (in particular, do not ready a G), as this can deadlock
	// with stack shrinking.
	lock mutex
}

type waitq struct {
	first *sudog
	last  *sudog
}

type sudog struct {
	// The following fields are protected by the hchan.lock of the
	// channel this sudog is blocking on. shrinkstack depends on
	// this for sudogs involved in channel ops.

	g *g

	// isSelect indicates g is participating in a select, so
	// g.selectDone must be CAS'd to win the wake-up race.
	isSelect bool
	next     *sudog
	prev     *sudog
	elem     unsafe.Pointer // data element (may point to stack)

	// The following fields are never accessed concurrently.
	// For channels, waitlink is only accessed by g.
	// For semaphores, all fields (including the ones above)
	// are only accessed when holding a semaRoot lock.

	acquiretime int64
	releasetime int64
	ticket      uint32
	parent      *sudog // semaRoot binary tree
	waitlink    *sudog // g.waiting list or semaRoot
	waittail    *sudog // semaRoot
	c           *hchan // channel
}

# buf 满

当 channel 满了活着没有 buf 的时候,我们继续 send,recv 时会阻塞当前 goroutine.

# 发送 g1 阻塞

阻塞操作是在 send 或 recv 时主动触发的.正在运行的 g1 运行 send 操作时会主动调用 GO 的调度器,让 g1 等待,让出 M,然后其他 g 去执行.同时 g1 也会被抽象成含有 g1 的指针和 send 元素的 sudog 结构体保存到 hchan 的 sendq 中等待被唤醒.

# 接收 g2 唤醒

当接收的 g2 执行 recv 时,g2 从缓存中取出数据,channel 会将等待队列中的 g1 推出,将 g1 当时 send 的数据推到 buf 中,然后调用 go 的 scheduler,唤醒 g1,并发 g1 放到可运行的 groutine 队列中.

# 先执行 g2

g2 会主动调用 go 调度器,让 g2 等待,并让出 M 让其他 g 去执行,g2 还会被抽象成含有 g2 指针和 recv 空元素的 sudog 结构体保存到 hchan 的 recvq 中等待被唤醒.此时恰好有个 g1 开始想 channel 发送数据,此时,g1 并不会锁住 channel 然后将数据放入缓存 buf 中而是直接将数据从 g1 copy 到 g2 的栈中,在唤醒过程中 g2 无需再获得 channel 的锁,再 copy 数据,减少了内存 copy 提升了效率.

# 通信

package main

import (
	"fmt"
	"sync"
)

func main() {
	fmt.Println(c(10))
	fmt.Println(m(10))
}

func c(n int) []int {
	var ints []int
	ch := make(chan int, n)
	for i := 0; i < n; i++ {
		go func(ch chan<- int, i int) {
			ch <- i
		}(ch, i)
	}
	for i := range ch {
		ints = append(ints, i)
		if len(ints) == n {
			break
		}
	}
	close(ch)
	return ints
}

func m(n int) []int {
	var ints []int
	var wg sync.WaitGroup
	var mux sync.Mutex
	wg.Add(n)
	for i := 0; i < n; i++ {
		go func(i int) {
			defer wg.Done()
			// 加锁,或者main中设置 runtime.GOMAXPROCS(1)
			mux.Lock() //
			ints = append(ints, i)
			mux.Unlock()
		}(i)
	}
	wg.Wait()
	return ints
}

# 阻塞

  1. 向 nil 通道发送数据会阻塞

  2. 向无缓存通道写数据,如果读协程没有准备好,会阻塞

  3. 向有缓存通道写数据,如果缓存区满了,会阻塞

  4. range 没有关闭的 channel ,也没有可读数据,会阻塞,close 时会 range 完所有数据后退出

  5. 从 nil 通道接收数据会阻塞

  6. 从无缓存读取数据,如果写协程没准备好会阻塞

  7. 从有缓存读取数据,如果为空会阻塞

# panic

  1. close 掉的 channel 写数据会发生 panic
  2. close 已经 close 的 channel 会 panic

读取的 channel 如果关闭了,不会影响正在读的数据,会将所有数据读取完毕,并不会立即失败或返回零值

无缓存通道不会使用 buf,有缓存的 channel 也不会每次发送和接收都经过 buf,如果发送的时候刚好有等待接收的协程,会直接交换数据.

channel 关闭最好在写入方关闭.

如果多个写协程的 channel 要关闭,可以使用额外 channel 来标记,也可以使用sync.Once和sync.Mutex来处理

# select

  1. 有 default 不会被阻塞,没有 default 没有可读可写 channel 会阻塞
  2. nil channel 读写都会阻塞
  3. select 不会向 range 一样发现 channel 被关闭而终止执行
  4. 如果多个 case 同时满足条件,会伪随机一个 case 执行
  5. select 不配合 for 只会对 case 表达式执行一次
  6. 每次 select 的执行都会扫描所有 case,而不是遇到合适的 case 就执行

# 斐波拉契数列

func main() {
	ch := make(chan int)
	q := make(chan int)
	go func(ch, q chan int) {
		for i := 0; i < 10; i++ {
			num := <-ch
			fmt.Println(num)
		}
		q <- 1
	}(ch, q)
	fibonacci := func(ch, q chan int) {
		x, y := 0, 1
		for {
			select {
			case ch <- x:
				x, y = y, x+y
			case <-q:
				fmt.Println("quit")
				return
			}
		}
	}
	fibonacci(ch, q)
}

# 例子

func sum(s []int, c chan int) {
	sum := 0
	for _, v := range s {
		sum += v
	}
	c <- sum
}

func main() {
	s := []int{7, 2, 8, -9, 4, 0}
	c := make(chan int)
	go sum(s[:len(s)/2], c)
	go sum(s[len(s)/2:], c)
	x, y := <-c, <-c // 会等待计算结果
	fmt.Println(x, y, x+y)
}

# 使用缓存

  • 有缓存通道是异步的,无缓存通道是同步的
  • 有缓存通道可供多个协程同时处理,可提高并发性
ch1 := make(chan int)
ch2 := make(chan int, 0)
ch3 := make(chan int, 1)
func test() {
  inCh := generator(100)
  outCh := make(chan int, 10)
  for i := 5 ; i<5 ; i++ {
    go do(inCh, outCh)
  }
  for r := range outCh {
    fmt.Println(r)
  }
}
func do(inCh <-chan int, outCh chan<- int) {
  for v := range inCh {
    outch<- v*v
  }
}

func doWithTimeOut(timeout time.Duration) (int, error) {
  select {
    case ret := <-do():
      return ret, nil
    case <-time.After(timeout):

}

# time 实现无阻塞读写

func unBlockRead(ch chan int) (x int, err error) {
	select {
	case x = <-ch:
		return x, nil
	case <-time.After(time.Microsecond):
		return 0, errors.New("read time out")
	}
}

func unBlockWrite(ch chan int, x int) (err error) {
	select {
	case ch <- x:
		return nil
	case <-time.After(time.Microsecond):
		return errors.New("read time out")
	}
}

func isChanClose(ch chan int) bool {

	case _, received := <-ch:
		return !received
	default:
	}

}

# panic recover

func SafeClose(ch chan int) (closed bool) {
	defer func() {
		if recover() != nil {
			closed = false
		}
	}()
	close(ch)
	return true
}

# 小结

  • channel 的发送接收操作本质上是值的拷贝
  • channel 超出作用范围时会被 gc 回收
  • 只读 channel 不能关闭
  • 最佳实践是使用无缓冲 channel,仅在需要时使用缓存 channel
